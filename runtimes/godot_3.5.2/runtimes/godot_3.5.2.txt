#!/bin/bash

GODOT_VERSION="3.5.2"
RUNTIME_DIR="$controlfolder/runtimes/godot_$GODOT_VERSION"

if [ -f "$RUNTIME_DIR/$DEVICE_ARCH.txt" ]; then
    source "$RUNTIME_DIR/$DEVICE_ARCH.txt"
else
    echo "GODOT $GODOT_VERSION: Unable to find a suitable runtime arch file."
    exit 255
fi

GODOT_RUN="$ESUDO $RUNTIME_DIR/$GODOT_BINARY $GODOT_OPTS"
